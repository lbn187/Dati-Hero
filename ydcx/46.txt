function　rSum（j：integer）：integer；
var
　　　sum：integer；
begin
　　　sum：=0；
　　　while　j<>0　do
　　　begin
　　　　　　sum：=sum*10+（j　mod　10）；
　　　　　　j：=j　div　10；
　　　end；
　　　rSum：=sum；
end；　
begin
　　　readln（n，m）；
　　　for　i：=n　to　m　do
　　　　　　if　i=rSum（i）
　　　　　　　　　then　write（I，' '）；
end.
